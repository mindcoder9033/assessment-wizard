\section*{2. Section A – Interpretation and Analysis (9 Marks)}

\subsection*{Question 1 – Interpretation of Graphs (3 Marks)}

\textbf{Question Type:} Short response  

\textbf{Requirement:}
\begin{itemize}
    \item Identify one issue that limits interpretation of the data in a given source.
    \item Clearly explain why this issue restricts interpretation.
\end{itemize}

\textbf{Skills Assessed:}
\begin{itemize}
    \item Understanding of data presentation
    \item Critical evaluation of graphical information
    \item Clarity of explanation
\end{itemize}

\textbf{Marking Overview}

\begin{center}

\toprule
\textbf{Marks} & \textbf{Performance Description} \\
\midrule
1 & Relevant issue identified \\
2 & Relevant issue clearly described \\
3 & Relevant issue fully explained \\
\bottomrule
\end{tabular}
\end{center}

\vspace{1em}

\subsection*{Question 2 – Data Analysis (6 Marks)}

\textbf{Question Type:} Analytical short response  

\textbf{Requirement:}
\begin{itemize}
    \item Analyze findings from one provided source.
    \item State a clear conclusion.
    \item Explicitly link the conclusion to the data.
\end{itemize}

\textbf{Expectations:}
\begin{itemize}
    \item Accurate interpretation of results
    \item Logical reasoning
    \item Clear connection between findings and conclusion
\end{itemize}

\textbf{Marking Overview}

\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Marks} & \textbf{Performance Description} \\
\midrule
1--2 & Limited or inaccurate analysis; unclear conclusion \\
3--4 & Accurate analysis but lacks development; basic conclusion \\
5--6 & Detailed analysis; clear and explicitly linked conclusion \\
\bottomrule
\end{tabular}
\end{center}

\section*{3. Section B – Research Considerations (6 Marks)}

\subsection*{Question 3 – Qualitative Research Considerations}

\textbf{Question Type:} Focused evaluative discussion  

\textbf{Possible Focus Areas:}
\begin{itemize}
    \item Improving credibility
    \item Avoiding bias
    \item Assessing transferability
\end{itemize}

\textbf{Requirement:}

\textbf{Skills Assessed:}

    \item Evaluation of research quality

\toprule
\textbf{Marks} & \textbf{Performance Description} \\
\midrule
1--2 & Limited understanding; little reference to source \\
3--4 & Some understanding; implicit reference to source \\
5--6 & Detailed understanding; explicit reference to source \\
\bottomrule
\end{tabular}

\section*{4. Section C – Synthesis and Evaluation (15 Marks)}

\subsection*{Question 4 – Extended Response}

\textbf{Question Type:} Extended analytical essay  

\textbf{Requirement:}
\begin{itemize}
    \item Use at least three sources.
    \item Apply relevant HL extension knowledge.
    \item Discuss the validity of a stated claim.
    \item Consider different perspectives.
    \item Reach a reasoned conclusion.
\end{itemize}

\textbf{Structure of a Strong Response:}

\textbf{Introduction}
\begin{itemize}
    \item Clear understanding of the claim
    \item Identification of relevant sources
\end{itemize}

\textbf{Analysis of Sources}
\begin{itemize}
    \item Interpretation of findings
    \item Integration of HL extension knowledge
    \item Clear linkage to the claim
\end{itemize}

\textbf{Evaluation}
\begin{itemize}
    \item Consideration of strengths and limitations
    \item Discussion of validity
    \item Consideration of alternative viewpoints
\end{itemize}

\textbf{Conclusion}
\begin{itemize}
    \item Clearly reasoned judgment
    \item Consistent with arguments presented
\end{itemize}

\textbf{Marking Overview}

\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Marks} & \textbf{Performance Description} \\
\midrule
1--3 & Mostly descriptive; minimal understanding \\
4--6 & Limited analysis; simplistic conclusion \\
7--9 & Some analysis; partially developed discussion \\
10--12 & Good critical analysis; clear argument \\
13--15 & Detailed, well-developed analysis; effective synthesis of at least three sources; reasoned conclusion \\
\bottomrule
\end{tabular}
\end{center}
