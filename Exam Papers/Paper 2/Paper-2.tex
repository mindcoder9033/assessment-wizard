\section*{2. Section A – Class Practicals (20 Marks)}

\subsection*{Overview}

\begin{itemize}
    \item Four compulsory questions
    \item Based on class practical investigations
    \item Focus on research method (not broader topic area)
\end{itemize}

\textbf{Assessment Objectives Assessed:}
\begin{itemize}
    \item AO1: Knowledge and understanding
    \item AO2: Application and analysis
    \item AO3: Synthesis and evaluation
\end{itemize}

% ------------------------------------------------
\subsection*{Question 1a – Knowledge and Understanding (4 marks)}

\textbf{Requirement}
\begin{itemize}
    \item Demonstrate knowledge of the research method used
    \item Describe the aim, procedure, and implementation
\end{itemize}

\textbf{High-Level Responses Include}
\begin{itemize}
    \item Accurate explanation of methodology
    \item Clear structure
    \item Precise terminology
\end{itemize}

% ------------------------------------------------
\subsection*{Question 1b – Application of a Concept (4 marks)}

\textbf{Requirement}
\begin{itemize}
    \item Apply one psychological concept to the practical
    \item Show influence on design, methodology, analysis, or findings
\end{itemize}

\textbf{Possible Concepts:}
Bias, Causality, Change, Measurement, Perspective, Responsibility

\textbf{High-Level Responses Include}
\begin{itemize}
    \item Clear concept explanation
    \item Direct links to the practical
    \item Analytical writing
\end{itemize}

% ------------------------------------------------
\subsection*{Question 1c – Compare and Contrast Methods (6 marks)}

\textbf{Requirement}
\begin{itemize}
    \item Compare method used with another research method
    \item Identify similarities and differences
\end{itemize}

\textbf{Expectations}
\begin{itemize}
    \item Balanced discussion
    \item At least two similarities and two differences
    \item Use of research terminology
\end{itemize}

% ------------------------------------------------
\subsection*{Question 1d – Research Design (6 marks)}

\textbf{Requirement}
\begin{itemize}
    \item Design a new study on the same topic
    \item Use a different research method
    \item Outline aim, procedure, and design elements
\end{itemize}

\textbf{High-Level Responses Include}
\begin{itemize}
    \item Logical procedure
    \item Feasible design
    \item Accurate terminology
\end{itemize}

\section*{3. Section B – Evaluation of an Unseen Study (15 Marks)}

\subsection*{Overview}

\begin{itemize}
    \item One extended response question
    \item Based on an unseen research study
    \item Discuss at least two specified concepts
\end{itemize}

\textbf{Assessment Objectives Assessed:}
\begin{itemize}
    \item AO1: Knowledge and understanding
    \item AO3: Synthesis and evaluation
\end{itemize}

% ------------------------------------------------
\subsection*{Nature of the Stimulus Study}

The study:
\begin{itemize}
    \item May be experimental or non-experimental
    \item Aligns with contexts such as:
    \begin{itemize}
        \item Health and well-being
        \item Human development
        \item Human relationships
        \item Learning and cognition
    \end{itemize}
\end{itemize}

\textbf{Possible Concepts:}
Bias, Causality, Change, Measurement, Perspective, Responsibility

% ------------------------------------------------
\subsection*{Question Requirements}

Students must:
\begin{enumerate}
    \item Demonstrate understanding of at least two concepts
    \item Apply concepts to the study
    \item Provide critical evaluation
    \item Reference stimulus material
    \item Reach a reasoned conclusion
\end{enumerate}

% ------------------------------------------------
\subsection*{Structure of a High-Quality Response}

\textbf{Introduction}
\begin{itemize}
    \item Brief overview of study
    \item Identification of chosen concepts
\end{itemize}

\textbf{Concept Discussion}
\begin{itemize}
    \item Definition
    \item Application
    \item Strengths and/or limitations
\end{itemize}

\textbf{Critical Analysis}
\begin{itemize}
    \item Validity
    \item Reliability
    \item Ethics
    \item Generalisability
\end{itemize}

\textbf{Conclusion}
\begin{itemize}
    \item Reasoned summary
    \item Logical linkage to arguments
\end{itemize}
